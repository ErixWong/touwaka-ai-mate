## 🔧 自动化检查建议

### 1. 添加i18n检查脚本

创建 `scripts/check-i18n.js`:

```javascript
const zh = require('../frontend/src/i18n/locales/zh-CN.ts');
const en = require('../frontend/src/i18n/locales/en-US.ts');

function compareKeys(obj1, obj2, path = '') {
  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);
  
  const onlyIn1 = keys1.filter(k => !keys2.includes(k));
  const onlyIn2 = keys2.filter(k => !keys1.includes(k));
  
  if (onlyIn1.length || onlyIn2.length) {
    console.log(`差异路径: ${path || 'root'}`);
    onlyIn1.forEach(k => console.log(`  只在中文: ${k}`));
    onlyIn2.forEach(k => console.log(`  只在英文: ${k}`));
  }
  
  // 递归检查对象
  keys1.forEach(k => {
    if (typeof obj1[k] === 'object' && typeof obj2[k] === 'object') {
      compareKeys(obj1[k], obj2[k], `${path}.${k}`);
    }
  });
}

compareKeys(zh, en);
```

### 2. 提交前检查

在 `package.json` 中添加：

```json
{
  "scripts": {
    "check:i18n": "node scripts/check-i18n.js"
  }
}
```
